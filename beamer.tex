\newcommand{\figplot}[3]{
	\begin{figure}
	\centering
	\input{plot/out/#1}
	\ifthenelse{\equal{#2}{}}{}{\caption{#2}}
	\ifthenelse{\equal{#3}{}}{}{\label{#3}}
	\end{figure}
}

	\begin{figure}
	\centering
	\includegraphics[#1]{#2}
	\ifthenelse{\equal{#3}{}}{}{\caption{#3}}
	\ifthenelse{\equal{#4}{}}{}{\label{#4}}
	\end{figure}
}
